# ══════════════════════════════════════════════════
#  Job Scheduling — Greedy Algorithm (DAA)
#  Greedy Choice: highest-profit job first
#  Time: O(n²)  |  Space: O(n)
# ══════════════════════════════════════════════════

def job_sequencing(jobs):
    """
    Maximize total profit by scheduling jobs
    before their respective deadlines.

    jobs: list of dict {id, deadline, profit}
    returns: (scheduled, timeline, total_profit)
    """

    # Step 1 — Sort by profit descending (greedy key)
    sorted_jobs = sorted(
        jobs,
        key=lambda j: j['profit'],
        reverse=True
    )

    max_d  = max(j['deadline'] for j in jobs)
    slot   = [None] * (max_d + 1)   # 1-indexed
    done   = []
    profit = 0

    # Step 2 — Greedy assignment
    for job in sorted_jobs:
        # Scan latest free slot ≤ deadline
        for t in range(job['deadline'], 0, -1):
            if slot[t] is None:
                slot[t] = job['id']
                done.append(job)
                profit += job['profit']
                break

    return done, slot[1:], profit


# ── Example run ────────────────────────────────
if __name__ == '__main__':
    jobs = [
        {'id':'J1', 'deadline':2, 'profit':100},
        {'id':'J2', 'deadline':1, 'profit': 80},
        {'id':'J3', 'deadline':2, 'profit': 60},
        {'id':'J4', 'deadline':1, 'profit': 40},
        {'id':'J5', 'deadline':3, 'profit': 20},
    ]
    sched, timeline, total = job_sequencing(jobs)
    print(f"Scheduled : {[j['id'] for j in sched]}")
    print(f"Timeline  : {timeline}")
    print(f"Profit    : ₹{total}")
    # Scheduled : ['J1', 'J2', 'J5']
    # Timeline  : ['J2', 'J1', 'J5']
    # Profit    : ₹200
